AuthMe 配置

第一次运行 AuthMe 的时候插件将会在 plugins/AuthMe 创建一份 config.yml， 你可以用其来配置各种设置。以下内容为 config.yml 的初始内容：

# Translated By CH1
DataSource:

    # 数据库类型: SQLITE(默认), MYSQL
    backend: 'SQLITE'

    caching: true
    # 数据库地址
    mySQLHost: '127.0.0.1'
    # 数据库端口
    mySQLPort: '3306'
    # 使用SSL(更安全)来连接数据库
    mySQLUseSSL: true

    mySQLUsername: 'authme'
    # 数据库密码
    mySQLPassword: '12345'

    mySQLDatabase: 'authme'
    # 数据表名称

        DenyTabCompleteBeforeLogin: false
        # 是否展示该玩家IP下所有账号
        # 权限 authme.admin.accounts
        displayOtherAccounts: true

        # IP最大登录玩家数量

        maxJoinPerIp: 0

        noTeleport: false
        # 密码允许的内容
        allowedPasswordCharacters: '[\x21-\x7E]*'
        # 其他关于账户命令使用阈值 小于2为禁用

    unrestrictions:

        # 例子:
        # UnrestrictedName:
        # - 'n**layer'

    security:
        # 密码最小位数
        minPasswordLength: 5
        # 密码最大位数
        passwordMaxLength: 30
        # 加密方法: SHA256, BCRYPT, BCRYPT2Y, PBKDF2, SALTEDSHA512, WHIRLPOOL,

        # 如果密码错误，会用以下方式来解密密码

        pbkdf2Rounds: 10000
        # 弱密码
